CREATE TABLE MEMBER( -- 회원가입 테이블
    ID VARCHAR2(30),
    PW VARCHAR2(30),
    NAME VARCHAR2(30),
    EMAIL VARCHAR2(30),
    TEL VARCHAR2(30)
);

CREATE SEQUENCE SEQ_LIST_NUMBER
INCREMENT BY 1
START WITH 1; -- 게시글 리스트 시퀀스


CREATE TABLE LIST( -- 게시글 리스트
    TITLE VARCHAR2(80),
    COMMENTS VARCHAR2(10),
    TIME DATE,
    NICNAME VARCHAR2(30),
    NUM NUMBER NOT NULL
);

CREATE TABLE WRITE_FORM( -- 게시글 폼(제목,내용)
    TITLE VARCHAR2(80),
    CONTENT VARCHAR2(1500)
);    

CREATE TABLE USER_COMMENT( -- 게시물 댓글
    LIST_NUM NUMBER,
    COMMENT_NUM NUMBER,
    NICNAME VARCHAR(30),
    post_COMMENT VARCHAR2(200),
    TIME DATE
);  
-- 댓글 시퀀스
CREATE SEQUENCE SEQ_USER_COMMENT;
